create table League (
	lid int,
	prize_pool int,
	primary key (lid)
);

create table Item (
	iid int,
	price int,
	primary key (iid)
);

create table Hero (
	hid int,
	ability varchar(100),
	primary key(hid)
);

create table Pro_Team (
	ptid int,
	primary key(ptid)
);

create table Team_Compose (
	mid int references Match_Host,
	radiant boolean,
	fb boolean,
	win boolean,
	tower int,
	primary key (mid, radiant)
);

create table Play_In (
	mid int,
	radiant boolean,
	ptid int references Pro_Team (ptid),
	primary key (mid, ptid),
	foreign key (mid, radiant) references Team_Compose
);

create table Pick (
	pid int references Player_Consist_Of (pid),
	hid int references Hero (hid),
	primary key (pid, hid)
);

create table Player_Consist_Of (
	pid int,
	position int,
	hero_used int,
	ptid int not null,
	foreign key (ptid) references Pro_Team (ptid),
	primary key (pid),
	check (position > 0 and position < 6)
);

create table Own (
	iid int references Item,
	hid int references Hero,
	bought_time time,
	primary key (iid, hid)
);

create table Belong_To (
	mid int,
	radiant boolean,
	hid int references Hero,
	last_hit int,
	death int,
	kill int,
	level int,
	denies int,
	gpm int,
	assist int,
	xpm int,
	primary key(mid, hid),
	foreign key (mid, radiant) references Team_Compose,
	check (level > 0 and level < 26)
);

create table Participate (
	lid int references League,
	ptid int references Pro_Team,
	rank int,
	win_rate int,
	primary key (lid, ptid),
	check (win_rate >= 0 and win_rate <= 100)
);

create table Match_Host (
	mid int,
	lid int not null,
	date date,
	duration time,
	primary key (mid),
	foreign key (lid)  references League 
);
